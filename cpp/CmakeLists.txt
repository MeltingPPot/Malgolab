cmake_minimum_required(VERSION 3.10)
project(MalgolabCpp)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release)
endif()

include_directories(${PROJECT_SOURCE_DIR}/include)

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${PROJECT_SOURCE_DIR}/build/bin)

file(GLOB APP_SOURCES "${PROJECT_SOURCE_DIR}/apps/*.cpp")

foreach(APP_SOURCE ${APP_SOURCES})

    get_filename_component(APP_NAME ${APP_SOURCE} NAME_WE)    
    add_executable(${APP_NAME} ${APP_SOURCE})

endforeach()

# 将 include 目录添加到头文件搜索路径中。
# 自动收集 apps 目录下所有的 .cpp 文件，存入 APP_SOURCES 变量。
# GLOB 会在配置时展开文件列表，如果新增文件，需要重新运行 CMake 才能检测到。
# 遍历每个源文件，为每个文件创建一个可执行目标。
# 获取文件名（不含扩展名），例如 segment_tree_cli.cpp → segment_tree_cli
# 添加一个可执行目标，目标名就是文件名，源文件为当前遍历的 .cpp 文件。
